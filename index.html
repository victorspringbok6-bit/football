async function processManualInput() { if (!apiKey) { alert("Please provide an API Key first."); return; } const txt = document.getElementById('manualInput').value.trim(); if (!txt) return; const tzoffset = (new Date()).getTimezoneOffset() * 60000; const todayStr = (new Date(Date.now() - tzoffset)).toISOString().split('T')[0]; // 1. Split by lines or common separators to get individual raw team/match lines const lines = txt.split(/\n|,/).map(s => s.trim()).filter(s => s.length > 2); if (lines.length === 0) return; document.getElementById('emptyState').classList.add('hidden'); document.getElementById('actionBar').classList.remove('hidden'); // Clean up words that throw off the search endpoint const cleaningRegex = /\b(nk|fc|afc|ud|cd|sc|ac|cs|fk|as|sv|bsc|cf|sd)\b/gi; for (let line of lines) { // 2. Extract clean search terms (e.g., "Aluminij Kidricevo" -> "Aluminij") let searchTerms = line.split(/\bvs\b|\bv\b/i); for (let term of searchTerms) { let cleanQuery = term.replace(cleaningRegex, '').replace(/[^a-zA-Z0-9 ]/g, '').trim(); // Grab the first significant word if it's a long multi-word string let words = cleanQuery.split(/\s+/).filter(w => w.length > 2); if (words.length === 0) continue; // Use the strongest keyword identifier for the API search let primarySearch = words[0]; const lowerSearch = primarySearch.toLowerCase(); const alreadyExists = allFixtures.some(f => f.teams.home.name.toLowerCase().includes(lowerSearch) || f.teams.away.name.toLowerCase().includes(lowerSearch) ); if (alreadyExists) continue; try { const response = await fetch(`https://v3.football.api-sports.io/fixtures?date=${todayStr}&search=${encodeURIComponent(primarySearch)}`, { "method": "GET", "headers": { "x-apisports-key": apiKey } }); const resData = await response.json(); if (resData.response && resData.response.length > 0) { resData.response.forEach(matchedFixture => { const isDuplicate = allFixtures.some(f => f.fixture.id === matchedFixture.fixture.id); if (!isDuplicate && matchedFixture.fixture.status.short === 'NS') { allFixtures.push(matchedFixture); unscannedFixtureIds.push(matchedFixture.fixture.id); } }); } } catch (e) { console.error("Error looking up team line item: ", primarySearch); } } } document.getElementById('manualInput').value = ""; renderAllFixtures(); updateBatchButtonText(); updateFilterCounters(); }
